//自然数的拆分
#include <iostream.h>
#include <iomanip.h>
void sum(int);
int a[20];
int i,j;
int main()
{
    cout<<"Input n:";cin>>a[0];
j=0;
if(a[0]>1)
{
    for(i=1;i<=a[0]/2;i++)
{
    a[1]=i;
a[2]=a[0]-a[1];
sum(2);
}
}
cout<<"Program End.";
cin>>i;
return 0;
}
void sum(int kx)
{
    int k,m,l;
j=j+1;
cout<<"Sum No."<<setw(3)<<j<<":"<<a[0]<<"=";
for(k=1;k<kx;k++)
    cout<<a[k]<<"+";
cout<<a[kx]<<endl;
k=kx;l=a[k];
for(m=a[k-1];m<=l/2;m++)
{
    a[k]=m;a[k+1]=l-m;
    sum(k+1);
}
}
